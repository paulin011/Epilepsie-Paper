import os

# List of required packages for the project
requirements = [
    "watchdog",  # For monitoring the directory for new PDF files
    "PyPDF2",    # For reading PDF files
    "markdown",   # For converting text to Markdown format
]

# Write the requirements to the requirements.txt file
with open(os.path.join("papers-to-md", "requirements.txt"), "w") as f:
    for requirement in requirements:
        f.write(requirement + "\n")
